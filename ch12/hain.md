```js
function map(arr, callback) {
  const result = [];

  for (let i = 0; i < arr.length; i++) {
    result.push(callback(arr[i]));
  }

  return result;
}
```

```js
function filter(arr, callback) {
  const result = [];

  for (let i = 0; i < arr.length; i++) {
    if (callback(arr[i])) {
      result.push(arr[i]);
    }
  }

  return result;
}
```

```js
function reduce(arr, callback, initValue) {
  let result = initValue;

  for (let i = 0; i < arr.length; i++) {
    result = callback(result, arr[i]);
  }

  return result;
}
```
